@Composable
fun HomeScreen(
    navController: NavController,
    viewModel: HomeViewModel
) {
    val palettes by viewModel.palettes.collectAsState()
    val colors by viewModel.colors.collectAsState()
    val isLoading by viewModel.isLoading.collectAsState()
    var selectedTab by remember { mutableIntStateOf(0) }

    LaunchedEffect(Unit) {
        viewModel.loadData()
    }

    Scaffold(
        topBar = {
            TabRow(selectedTabIndex = selectedTab) {
                Tab(
                    selected = selectedTab == 0,
                    onClick = { selectedTab = 0 },
                    text = { Text("Цвета") }
                )
                Tab(
                    selected = selectedTab == 1,
                    onClick = { selectedTab = 1 },
                    text = { Text("Палитры") }
                )
            }
        },
        floatingActionButton = {
            FloatingActionButton(
                onClick = {
                    val mode = if (selectedTab == 0) "color" else "palette"
                    navController.navigate(Screen.Camera.createRoute(mode))
                },
                modifier = Modifier.navigationBarsPadding()
            ) {
                Text(if (selectedTab == 0) "+" else "+")
            }
        }
    ) { padding ->
        when {
            isLoading -> {
                Box(
                    modifier = Modifier
                        .fillMaxSize()
                        .padding(padding),
                    contentAlignment = Alignment.Center
                ) {
                    CircularProgressIndicator()
                }
            }
            selectedTab == 0 -> {
                ColorListScreen(
                    colors = colors,
                    viewModel = viewModel,
                    modifier = Modifier.padding(padding)
                )
            }

            else -> {
                PaletteListScreen(
                    palettes = palettes,
                    viewModel = viewModel,
                    modifier = Modifier.padding(padding)
                )
            }
        }
    }
}

@Composable
fun ColorListScreen(
    colors: List<Color>,
    viewModel: HomeViewModel,
    modifier: Modifier
) {
    LazyColumn(
        modifier = modifier.fillMaxSize(),
        contentPadding = PaddingValues(bottom = 96.dp)
    ) {
        items(colors, key = { it.id }) { color ->
            ColorCard(color = color, viewModel = viewModel)
        }
    }
}

@Composable
fun ColorCard(
    color: Color,
    viewModel: HomeViewModel
) {
    var showDeleteDialog by remember { mutableStateOf(false) }

    Card(
        modifier = Modifier
            .fillMaxWidth()
            .padding(horizontal = 8.dp, vertical = 6.dp)
            .clickable { }
    ) {
        Row(
            modifier = Modifier
                .fillMaxWidth()
                .padding(10.dp),
            verticalAlignment = Alignment.CenterVertically,
            horizontalArrangement = Arrangement.SpaceBetween
        ) {
            Row(verticalAlignment = Alignment.CenterVertically) {
                Box(
                    modifier = Modifier
                        .size(60.dp)
                        .background(ComposeColor(android.graphics.Color.parseColor(color.hex)))
                )
                Spacer(modifier = Modifier.width(8.dp))
                Column {
                    Text(color.hex, fontFamily = FontFamily.Monospace)
                    Text(
                        "RGB: ${color.rgb.first}, ${color.rgb.second}, ${color.rgb.third}",
                        fontSize = 12.sp
                    )
                }
            }

            IconButton(onClick = { showDeleteDialog = true }) {
                Icon(Icons.Default.Delete, contentDescription = "Удалить")
            }
        }
    }

    if (showDeleteDialog) {
        AlertDialog(
            onDismissRequest = { showDeleteDialog = false },
            title = { Text("Удалить цвет") },
            text = { Text("Вы уверены, что хотите удалить этот цвет?") },
            confirmButton = {
                Button(
                    onClick = {
                        viewModel.deleteColor(color.id)
                        showDeleteDialog = false
                    }
                ) {
                    Text("Удалить")
                }
            },
            dismissButton = {
                Button(onClick = { showDeleteDialog = false }) {
                    Text("Отмена")
                }
            }
        )
    }
}
