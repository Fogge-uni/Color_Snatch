package com.example.testapp.domain.utils

import com.example.testapp.domain.model.Color
import kotlin.math.abs

interface ColorConverter {
    fun hexToRgb(hex: String): Triple<Int, Int, Int>
    fun rgbToCmyk(rgb: Triple<Int, Int, Int>): Color.CMYK
    fun rgbToHsl(rgb: Triple<Int, Int, Int>): Color.HSL
}

class ColorConverterImpl : ColorConverter {

    override fun hexToRgb(hex: String): Triple<Int, Int, Int> {
        val clean = hex.removePrefix("#")
        val intValue = clean.toLong(16)
        return Triple(
            ((intValue shr 16) and 0xFF).toInt(),
            ((intValue shr 8) and 0xFF).toInt(),
            (intValue and 0xFF).toInt()
        )
    }

    override fun rgbToCmyk(rgb: Triple<Int, Int, Int>): Color.CMYK {
        val r = rgb.first / 255.0
        val g = rgb.second / 255.0
        val b = rgb.third / 255.0

        val k = 1 - maxOf(r, g, b)
        if (k == 1.0) return Color.CMYK(0.0, 0.0, 0.0, 1.0)

        val c = (1 - r - k) / (1 - k)
        val m = (1 - g - k) / (1 - k)
        val y = (1 - b - k) / (1 - k)

        return Color.CMYK(c, m, y, k)
    }

    override fun rgbToHsl(rgb: Triple<Int, Int, Int>): Color.HSL {
        val r = rgb.first / 255.0
        val g = rgb.second / 255.0
        val b = rgb.third / 255.0

        val max = maxOf(r, g, b)
        val min = minOf(r, g, b)
        val delta = max - min

        val lightness = (max + min) / 2
        val saturation = if (delta == 0.0) 0.0 else delta / (1 - abs(2 * lightness - 1))

        var hue = when {
            delta == 0.0 -> 0.0
            max == r -> 60 * (((g - b) / delta) % 6)
            max == g -> 60 * (((b - r) / delta) + 2)
            else -> 60 * (((r - g) / delta) + 4)
        }

        if (hue < 0) hue += 360.0

        return Color.HSL(hue, saturation, lightness)
    }
}
